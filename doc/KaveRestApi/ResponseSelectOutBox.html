def entries(select=nil)
  if select.nil?
    return @response['entries']
  else
    return @response['entries'].select { |en|
    en['status'] == (KaveRestApi::ENTRIES[select].nil? ? select:KaveRestApi::ENTRIES[select])
    }
  end
end
